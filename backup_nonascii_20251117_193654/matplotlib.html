Python Code Example:

import matplotlib.pyplot as plt
import numpy as np

# Line plot example
x = np.linspace(0, 10, 100)
y = np.sin(x)
plt.plot(x, y, linewidth=2, label='sin(x)')

# Histogram example
data = np.random.normal(0, 1, 1000)
plt.hist(data, bins=30, color='orange', alpha=0.7)

# Scatter plot example
x_scatter = np.random.randn(100)
y_scatter = np.random.randn(100)
plt.scatter(x_scatter, y_scatter, c='red', alpha=0.5)

plt.xlabel('X Axis')
plt.ylabel('Y Axis')
plt.legend()
plt.show()
